Smart Contract (Solana Program)

// Rust Solana program for x402 distribution
#[program]
pub mod oasis_x402_distributor {
    pub fn distribute_payment(
        ctx: Context<DistributePayment>,
        nft_mint: Pubkey,
        total_amount: u64,
    ) -> Result<()> {
        // Get all NFT holders
        let holders = get_nft_holders(&nft_mint)?;
        
        // Calculate share per holder
        let amount_per_holder = total_amount / holders.len() as u64;
        
        // Transfer to each holder
        for holder in holders {
            transfer_sol(
                ctx.accounts.treasury.to_account_info(),
                holder.wallet.to_account_info(),
                amount_per_holder
            )?;
        }
        
        Ok(())
    }
}
